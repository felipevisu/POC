From Schema to Safety in One Line

import { z } from 'zod';
// 1. Define your schema
const UserSchema = z.object({
  id: z.number(),
  email: z.string().email(),
  age: z.number().min(18).max(120),
  isActive: z.boolean().default(true)
});

// 2. Get TypeScript type automatically
type User = z.infer;
// Result: { id: number; email: string; age: number; isActive: boolean; }

// 3. Validate runtime data
const result = UserSchema.parse(apiData);
// ‚úÖ Throws if invalid, returns typed data if valid

Notice: We never wrote the TypeScript interface manually!

const UserSchema = z.object({
    id: z.string().transform(Number),                           // string ‚Üí number
    name: z.string().transform(s => s.trim().toLowerCase()),    // clean whitespace
    email: z.string().email().transform(s => s.toLowerCase()), // normalize email
    age: z.string().transform(Number),                          // string ‚Üí number
    salary: z.string().transform(s => 
    parseFloat(s.replace(/[$,]/g, ''))                        // "$75,000.50" ‚Üí 75000.5
    ),
    tags: z.string().transform(s => 
    s.split(',').map(tag => tag.trim())                       // CSV ‚Üí string[]
    ),
    isActive: z.string().transform(s => s === 'true'),          // string ‚Üí boolean
    createdAt: z.string().transform(s => new Date(s))           // string ‚Üí Date
});
